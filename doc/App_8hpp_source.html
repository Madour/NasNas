301  template <typename T, typename... Targs, typename = std::enable_if<std::is_base_of_v<AppState, T>>>
302  void pushState(Targs... args) {
303  m_state_stack.emplace(std::make_unique<T>(std::forward<Targs>(args)...));
304  m_cb_onevent = [&](const sf::Event& event) { m_state_stack.top()->onEvent(event); };
305  m_cb_update = [&] { m_state_stack.top()->update(); };
306  m_cb_prerender = [&] { m_state_stack.top()->preRender(); };
307  }
308 
309  inline void popState() {
310  if (m_state_stack.empty())
311  return;
312 
313  m_state_stack.pop();
314 
315  if (m_state_stack.empty()) {
316  m_cb_onevent = [&](const sf::Event& event) {};
317  m_cb_update = [&] {};
318  m_cb_prerender = [&] {};
319  }
