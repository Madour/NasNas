312  template <typename T, typename... Targs, typename = std::enable_if<std::is_base_of_v<AppState, T>>>
313  void pushState(Targs... args) {
314  m_state_stack.emplace(std::make_unique<T>(std::forward<Targs>(args)...));
315  m_cb_onevent = [&](const sf::Event& event) { m_state_stack.top()->onEvent(event); };
316  m_cb_update = [&] { m_state_stack.top()->update(); };
317  m_cb_prerender = [&] { m_state_stack.top()->preRender(); };
318  }
319 
320  inline void popState() {
321  if (m_state_stack.empty())
322  return;
323 
324  m_state_stack.pop();
325 
326  if (m_state_stack.empty()) {
327  m_cb_onevent = [&](const sf::Event& event) {};
328  m_cb_update = [&] {};
329  m_cb_prerender = [&] {};
330  }
